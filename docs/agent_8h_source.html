50template<typename CellType>
51CellType* RandomMovePolicy<CellType>::selectCell(
52 fpmas::model::Neighbors<CellType> &mobility_field) const {
53 std::vector<float> utilities;
54 std::vector<CellType*> cells;
55
56 bool null_utilities = true;
57 for(auto cell : mobility_field) {
58 fpmas::model::ReadGuard read(cell);
59 if(cell->getUtility() > 0)
60 null_utilities = false;
61 utilities.push_back(cell->getUtility());
62 cells.push_back(cell);
63 }
64 std::size_t rd_index;
65 if(null_utilities) {
66 fpmas::random::UniformIntDistribution<std::size_t> rd_cell(0, cells.size()-1);
67 rd_index = rd_cell(fpmas::model::RandomNeighbors::rd);
68 } else {
69 fpmas::random::DiscreteDistribution<std::size_t> rd_cell(utilities);
70 rd_index = rd_cell(fpmas::model::RandomNeighbors::rd);
71 }
72 return cells[rd_index];
73}

75template<typename CellType>
76CellType* MaxMovePolicy<CellType>::selectCell(
77 fpmas::model::Neighbors<CellType> &mobility_field) const {
78 // Prevents bias when several cells have the max value
79 mobility_field.shuffle();
80
81 std::vector<std::pair<CellType*, float>> cells;
82 for(auto cell : mobility_field) {
83 fpmas::model::ReadGuard read(cell);
84 cells.push_back({cell, cell->getUtility()});
85 }
86
87 return std::max_element(cells.begin(), cells.end(),
88 [] (
89 const std::pair<CellType*, float>& a1,
90 const std::pair<CellType*, float>& a2
91 ) {
92 return a1.second < a2.second;
93 }
94 )->first;
95}

250template<typename AgentBase, typename PerceptionRange>
251void MetaAgent<AgentBase, PerceptionRange>::add_to_contacts(fpmas::api::model::Agent* agent) {
252 if(contacts().size() == max_contacts) {
253 for(auto edge : this->node()->getOutgoingEdges(CONTACT)) {
254 // Finds the edge corresponding to the queue's head and unlinks it
255 if(edge->getTargetNode()->getId() == contacts().front()) {
256 this->model()->graph().unlink(edge);
257 // No need to loop until the last edge
258 break;
259 }
260 }
261 // Removes queue's head once unlinked
262 contacts().pop_front();
263 }
264 // Links the new contact...
265 this->model()->link(this, agent, CONTACT)->setWeight(MetaAgentBase::contact_weight);
266 // ... and adds it at the end of the queue
267 contacts().push_back(agent->node()->getId());
268}

270template<typename AgentBase, typename PerceptionRange>
271void MetaAgent<AgentBase, PerceptionRange>::create_relations_from_neighborhood() {
272 // Agents currently in the Moore neighborhood
273 auto perceptions = this->perceptions();
274 // Shuffles perceptions, to ensure that a random perception will be
275 // selected as a new contact
276 perceptions.shuffle();
277
278 auto current_perception = perceptions.begin();
279 // Searches for an agent not already in this agent's contact in the current
280 // Moore neighborhood
281 while(current_perception != perceptions.end()) {
282 if(!is_in_contacts(current_perception->agent()->node()->getId())) {
283 add_to_contacts(*current_perception);
284
285 // Ends while loop
286 current_perception = perceptions.end();
287 } else {
288 current_perception++;
289 }
290 }
291}

293template<typename AgentBase, typename PerceptionRange>
294void MetaAgent<AgentBase, PerceptionRange>::create_relations_from_contacts() {
295 auto contacts =
296 this->template outNeighbors<MetaAgent<AgentBase, PerceptionRange>>(CONTACT);
297 if(contacts.count() >= 2) {
298 // Selects a random contact and puts it at the begining of the list.
299 std::size_t random_index =
300 fpmas::random::UniformIntDistribution<std::size_t>(0, contacts.count()-1)(
301 fpmas::model::RandomNeighbors::rd
302 );
303 std::swap(contacts[0], contacts[random_index]);
304 // contacts[0] is now the contact to which we would like to add a new
305 // contact from this agent's contacts.
306 std::size_t i = 1;
307 {
308 // Reads contacts[0], so that is_in_contacts() can be called safely
309 fpmas::model::ReadGuard read(contacts[0]);
310 // Searches a contact from this agent's contacts that is not
311 // already in contacts[0]'s contacts
312 while(
313 i < contacts.count() &&
314 contacts[0]->is_in_contacts(contacts[i]->node()->getId()))
315 i++;
316 }
317 if(i < contacts.count())
318 // If founds, creates a NEW_CONTACT, that will be handled from
319 // contacts[0] by handle_new_contacts() after the next
320 // synchronization.
321 this->model()->link(contacts[0], contacts[i], NEW_CONTACT);
322 }
323}

338template<typename AgentBase, typename PerceptionRange>
339void MetaAgent<AgentBase, PerceptionRange>::move() {
340 auto mobility_field = this->mobilityField();
341 typename AgentBase::Cell* selected_cell;
342 switch(move_policy) {
343 case MovePolicy::RANDOM:
344 selected_cell = RandomMovePolicy<typename AgentBase::Cell>()
345 .selectCell(mobility_field);
346 break;
347 case MovePolicy::MAX:
348 selected_cell = MaxMovePolicy<typename AgentBase::Cell>()
349 .selectCell(mobility_field);

351 };
